Alle Kräfte werden zerlegt in x und y Komponente:
\begin{align*}
  \mathbf{S_5} &=
  \begin{pmatrix} S_{5X} \\ S_{5Y}
  \end{pmatrix}
  =
  \begin{pmatrix} A_X \\ A_Y
  \end{pmatrix}
  =
  \begin{pmatrix} -16{,}422 \\ -3{,}556
  \end{pmatrix} \text{kN} \\[2ex]
  \mathbf{S_3} &=
  \begin{pmatrix} S_3 \cdot \cos(-60^\circ) \\ S_3 \cdot \sin(-60^\circ)
  \end{pmatrix}
  =
  \begin{pmatrix} 11{,}916 \cdot 0{,}5 \\ 11{,}916 \cdot \left(-\frac{\sqrt{3}}{2}\right)
  \end{pmatrix}
  =
  \begin{pmatrix} 5{,}958 \\ -10{,}320
  \end{pmatrix} \text{kN} \\[2ex]
  \mathbf{S_2} &=
  \begin{pmatrix} S_2 \cdot \cos(-120^\circ) \\ S_2 \cdot \sin(-120^\circ)
  \end{pmatrix}
  =
  \begin{pmatrix} -20{,}928 \cdot (-0{,}5) \\ -20{,}928 \cdot \left(-\frac{\sqrt{3}}{2}\right)
  \end{pmatrix}
  =
  \begin{pmatrix} 10{,}464 \\ 18{,}124
  \end{pmatrix} \text{kN}
\end{align*}

Wegen eine Belastung im Punkt D werden die Schnittgrößenfunktionen aus 2 Teile bestehen:
\begin{enumerate}
  \item $0 < x < a$
  \item $a < x < a + b$
\end{enumerate}
\subsection{Teil 1. (0 < x < a)}
\begin{align*}
  N(x) &= S_{5X} + S_{3X} = -10,464 \, \text{[kN]} \\
  V(x) &= S_{5Y} + S_{3Y} - p \cdot x = -13,876 - 0,5x \, \text{[kN]} \\
  M_b(x) &= (S_{5Y} + S_{3Y}) \cdot x + p \cdot \frac{x^2}{2} = 13{,}876x + 0{,}25x^2 \, \text{[kNm]}
\end{align*}
\subsection{Teil 2. (a < x < a + b)}
\begin{align*}
  N(x) &= 0 \, \text{[kN]} \\
  V(x) &= G + p \cdot (a + b - x) = 4,25 - 0,5x \, \text{[kN]}\\
  M_b(x) &= G \cdot (a + b - x) + p \cdot \frac{(a + b - x)^2}{2} = \frac{x^2}{4} - 4,25x + 9,0625 \, \text{[kNm]}
\end{align*}
